package PK;

import RTools;
import Tips;

public class ServiceException extends RuntimeException{

	private static final long serialVersionUID = 1L;


	public ServiceException(Tips tips) {
		super(RTools.custom(tips).toJSONString());
	}
	
	public ServiceException(Tips tips, String params) {
		super(RTools.custom(tips, params).toJSONString());
	}
	
	public ServiceException(Tips tips, String paramsFormat, Object... args) {
		super(RTools.custom(tips, String.format(paramsFormat, args)).toJSONString());
	}
	
	public ServiceException(String msg){
		this(0, msg);
	}
	
	public ServiceException(String msgFormat, Object... args) {
		this(0, String.format(msgFormat, args));
	}
	
	public ServiceException(int code, String msg){
		super(RTools.custom().putCode(code).putMsg(msg).toJSONString());
	}
	
	public ServiceException(int code, String msgFormat, Object... args) {
		super(RTools.custom().putCode(code).putMsg(String.format(msgFormat, args)).toJSONString());
	}

}
