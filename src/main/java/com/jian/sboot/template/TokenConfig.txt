package PK;

import org.springframework.boot.autoconfigure.condition.ConditionalOnMissingBean;
import org.springframework.context.annotation.Bean;
import org.springframework.stereotype.Component;

import com.jian.handle.TokenHandler;
import DefaultTokenHandler;

/**
 * 	注入token工具类
 * 
 * @author liujian
 */
@Component
public class TokenConfig {
	
	/**
	 * 	默认注入DefaultTokenHandler
	 * @return
	 */
	@Bean
	@ConditionalOnMissingBean(TokenHandler.class)
	public TokenHandler defTokenHandle() {
		System.out.println("注入默认的TokenHandler。。。");
		return new DefaultTokenHandler();
	}
	
}
